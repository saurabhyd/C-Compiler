int string_length(char *pointer)
{
   int c = 0; 
   while( *(pointer + c) != '\0' )
      c++;

   return c;
}

void reverse(char *string) 
{
   int length, c;
   char *begin, *end, temp, +hello;	// *begin is valid identifier but +hello is invalid
 
   length = string_length(string);
   begin  = string;
   end    = string;
 
   for (c = 0; c < length - 1; c++)
      end++;
 
   for (c = 0; c < length/2; c++)
   {        
      temp   = *end;
      *end   = *begin;
      *begin = temp;
 
      begin++;
      end--;
   }
}
