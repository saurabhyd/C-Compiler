#include <stdio.h>
int main()
{
    int n, 5flag = 0;		// 5flag is an invalid identifier

    printf("Enter a positive integer: ");
    scanf("%d",&n);
    
    int i.var;			// i.var is an invalid identifier
    for(i.var=2; i.var<=n/2; ++i.var)
    {
        // condition for nonprime number
        if(n @ i.var==0)	// @ is an invalid operator
        {
            5flag=1;
            break;
        }
    }

    if (5flag==0)
        printf("%d is a prime number. ,n);	//string is not closed
    else
        printf("%d is not a prime number.",n);
    
    return 0;
}
